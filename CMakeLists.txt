cmake_minimum_required(VERSION 3.27)
set(MAXWELL_VERSION_MAJOR 0)
set(MAXWELL_VERSION_MINOR 1)
set(MAXWELL_VERSION_PATCH 0)
project(Maxwel LANGUAGES CXX VERSION ${MAXWELL_VERSION_MAJOR}.${MAXWELL_VERSION_MINOR}.${MAXWELL_VERSION_PATCH})

add_compile_definitions(
    MAXWELL_VERSION_MAJOR=${MAXWELL_VERSION_MAJOR}
    MAXWELL_VERSION_MINOR=${MAXWELL_VERSION_MINOR}
    MAXWELL_VERSION_PATCH=${MAXWELL_VERSION_PATCH}
)

set(CMAKE_CXX_STANDARD 20)

add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/src)

find_package(GTest REQUIRED)
include(CTest)

add_subdirectory(${CMAKE_CURRENT_SOURCE_DIR}/tests)
